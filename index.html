Acerca de

En este solucionario se proponen 4 ejercicios relacionados con el uso de archivos XML, JSON y AJAX, de forma adicional se proporcionará y explicará las líneas de Código HTML, CSS y JavaScript necesarias para poder resolver los ejercicios propuestos de manera correcta

Ejercicios propuestos.

  • Ejercicio 1: Analizar una cadena de texto en un objeto DOM XML y extraer esa Información con JavaScript, para este ejercicio se utilizara una cadena de texto sobre animales pertenecientes a un zoloogico
  • Ejercicio 2: Desarrollar la función CargarDatos(), para desplegar el contenido (Asincrónico) de un archivo XML
  • Ejercicio 3: Escoja un actor de películas preferido para desplegar una tabla con la información correspondiente respsecto a un JSON
  • Ejercicio 4: Mostrar datos XML en una tabla utilizando AJAX

Soluciones

A continuación, se presentan las soluciones de cada uno de los ejercicios con loques de código que contienen os elementos necesarios para la correcta ejecución de los ejercicios, adicionalmente se explica la funcionalidad de las funciones y parten importantes de código JavaScript como HTML y los recursos tanto XML, JSON y AJAX cabe recalcar que no se aplicaran clases css en los bloques de codigo de explicación para evitar confuciones al momento de la implementación

Sección del scrip

La explicación de los procesos que hace el script están comentados dentro del mismo

// Desarrolle el script correspondiente, para que se escoja el actor y desplique todas las peliculas correspondientes a el.
function change_myselect(sel) {
  
  //obtiene el valor de la selección del despliegue de opciones 
  var obj, dbParam, xmlhttp;
  obj = { table: sel, limit: 20 };

  //Almacena una cadena de texto de formato JSON, transformando el objeto JS a JSON
  dbParam = JSON.stringify(obj);
  xmlhttp = new XMLHttpRequest();

  xmlhttp.onreadystatechange = function () {
    if (this.readyState == 4 && this.status == 200) {

      //Transforma a tipo JSON
      var myObj = JSON.parse(this.responseText);

      //Crea los encabezados de la tabla con un tipo de borde especifico 
      var table = "<table class='table'><thead '><tr><th>Genero</th><th>Nombre</th></tr></thead><tbody class='table-group-divider'>";
      var i;

      //A fin de iterar en cada objeto de las películas 
      for (i in myObj.peliculas) {

        //Busca en caso de ser igual a la selección realizada comparando con los protagonistas de los objetos JSON
        if (myObj.peliculas[i].Protagonista == sel) {
          table += "<tr><td>" + myObj.peliculas[i].Genero + "</td><td>" + myObj.peliculas[i].Nombre + "</td></tr>";
        }
      }

      //Se cierra la tabla, se busca el id demo y se reemplaza por el valor que lleva la variable table
      table += "</table>";
      document.getElementById("demo").innerHTML = table;
    }
  };

  xmlhttp.open("GET", "preg3.json", true);
  xmlhttp.setRequestHeader("Content-type", "application/x-www-form-urlencoded");
  xmlhttp.send("x=" + dbParam);
}

function CargarTabla() {
  var xhttp = new XMLHttpRequest();
  xhttp.onreadystatechange = function () {
    if (this.readyState == 4 && this.status == 200) {
      funcionTabla(this);
    }
  };

  xhttp.open("GET", "preg4.xml", true);
  xhttp.send();
}

function funcionTabla(xml) {
  var i;
  var xmlDoc = xml.responseXML;

  //Encabezado de la tabla a crear
  var table = "<tr><th>Numero</th><th>Title</th><th>Descripción</th><th>PrecioU</th><th>Cantidad</th><th>PrecioT</th></tr>";
  
  //Se obtienen todos los elementos que estén dentro de la etiqueta producto   var productos = xmlDoc.getElementsByTagName("PRODUCTO");
  for (i = 0; i < productos.length; i++) {

    //Para crear la tabla con cada producto nuevo se genera una fila nueva
    //La línea se repite por la cantidad de atributos existentes 
    table += "<tr><td>" +
      productos[i].getElementsByTagName("NUMERO")[0].childNodes[0].nodeValue +
      "</td><td>" +
      productos[i].getElementsByTagName("TITLE")[0].childNodes[0].nodeValue +
      "</td><td>" +
      productos[i].getElementsByTagName("DESCRIPCION")[0].childNodes[0].nodeValue +
      "</td><td>" +
      productos[i].getElementsByTagName("PRECIOU")[0].childNodes[0].nodeValue +
      "</td><td>" +
      productos[i].getElementsByTagName("CANTIDAD")[0].childNodes[0].nodeValue +
      "</td><td>" +
      productos[i].getElementsByTagName("PRECIOT")[0].childNodes[0].nodeValue +
      "</td></tr>";
  }

  //En todo el documento se busca el ID tabaventas para reemplazar por lo que se guardo dentro de la variable table 
  document.getElementById("tablaVentas").innerHTML = table;
}

function funcionTabla(xml) {
  var i;
  var xmlDoc = xml.responseXML;
  var table = "<tr><th>Numero</th><th>Title</th><th>Descripción</th><th>PrecioU</th><th>Cantidad</th><th>PrecioT</th></tr>";
  var productos = xmlDoc.getElementsByTagName("PRODUCTO");

  for (i = 0; i < productos.length; i++) {

      productos[i].getElementsByTagName("NUMERO")[0].childNodes[0].nodeValue +
